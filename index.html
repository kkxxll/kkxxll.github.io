es6map-set与数组和对象的比较

发表于 2017-08-06 | 分类于 es6 
{
  // map和array的对比
  // 数据结构横向对比，增，查，改，删
  let map=new Map();
  let array=[];
  // 增
  map.set('t',1);
  array.push({t:1});
  console.log('map-array',map,array);//{"t" => 1}  array[0]为{t:1}
  // 查
  let map_exist=map.has('t');
  let array_exist=array.find(item=>item.t);  //箭头函数，遍历item 如果存在't'，则返回item.t
  console.log('map-array',map_exist,array_exist); //true [{t:1}]  //第二个很奇怪应该是返回1
  // 改
  map.set('t',2);
  array.forEach(item=>item.t?item.t=2:'');
  console.log('map-array-modify',map,array); //{"t" => 2} [{t:2}]
  // 删
  map.delete('t');
  let index=array.findIndex(item=>item.t);
  console.log(index); //0
  array.splice(index,1); //删除第0项
  console.info('map-array-empty',map,array);//{} []
}

es6Symbol用法

发表于 2017-08-06 | 分类于 es6 
{
  // 声明
  let a1=Symbol();
  let a2=Symbol();
  console.log(a1===a2);   //false  说明Symbol()声明的变量是唯一的，独一无二的
  let a3=Symbol.for('kk'); //kk被Symbol()声明且存在a3里
  let a4=Symbol.for('kk'); //kk已经被Symbol()声明过了且存在a3里，所以不需要再声明，直接把a3赋值给a4
  let a5=Symbol.for('a5'); //同理
  let a6=Symbol.for('a5');
  console.log(a3===a4);  //true  
  console.log(a5===a6);  //true
}

es6Decorator

Decorator (修饰器) 是一个函数 用来修改类的行为

在.babelrc里面加插件 "plugins":["transform-decorators-legacy"]
{
  let readonly=function(target,name,descriptor){ //修饰器是一个函数
    descriptor.writable=false;
    return descriptor
  };
  class Test{
    @readonly  //对time进行只读
    time(){
      return '2017-03-11'
    }
  }

  // test.time=function(){

  // }; //这里会报错，因为time是只读的不允许修改
