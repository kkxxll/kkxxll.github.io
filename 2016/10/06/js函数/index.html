js函数

发表于 2016-10-06 | 分类于 js
1.函数可以作为变量
var one = function(){
    return 1;
}
2.函数可以作为数组里的一个元素
var two = [1,function(a){
    return a;
}]

two[1](5)//5

3.函数可以作为对象的成员变量
var three = {first: 1,fun: function(a){return a;}}

three.fun("hello") //hello

4.函数可以在使用时直接创建出来

10+(function(){return 1})() //11

5.函数可以作为参数被传递
function fn1(n,fn){
    return n+fn();
}

fn1(10,function(){return 1}); //11

6.函数可以被另一个函数返回
var a = function(){return function(){return 1}};
a() //function (){return 1}
a()() //1
