es6的set-map数据结构

发表于 2017-08-03 | 分类于 es6 
//不带参数的定义
{
  let list = new Set();
  list.add(5);
  list.add(7);
  console.log('size',list);//2 {5,7}
}
{
  let arr = [1,2,3,4,5];
  let list = new Set(arr);
  console.log('size',list);//5 {1,2,3,4,5}
}

//带参数的定义
{
  //数据唯一的，重复的会过滤掉，且数据类型不会自动转换 ，如'2'还是'2'
  let list = new Set();
  list.add(1);
  list.add(2);
  list.add(1);
  console.log('list',list); //{1,2}  
  let arr=[1,2,3,1,'2'];
  let list2=new Set(arr);
  console.log('unique',list2); //{1,2,3,'2'}  应用场景：去重
}

{
  let arr=['add','delete','clear','has'];
  let list=new Set(arr);
  console.log('has',list.has('add'));//true
  console.log('delete',list.delete('add'),list);//true {"delete", "clear", "has"}
  list.clear();
  console.log('list',list);//{}
}

{
  let arr=['add','delete','clear','has'];
  let list=new Set(arr);
//遍历
  for(let key of list.keys()){
    console.log('keys',key);
    //  keys add
    //  keys delete
    //  keys clear
    //  keys has
  }

  for(let value of list.values()){
    console.log('value',value);
    //  value add
    //  value delete
    //  value clear
    //  value has
  }
  for(let value of list){   //等价于ist.values()
    console.log('value',value);
    //  value add
    //  value delete
    //  value clear
    //  value has
  }
  for(let [key,value] of list.entries()){
    console.log('entries',key,value);
    //  entries add add
    //  entries delete delete
    //  entries clear clear
    //  entries has has
  }
  list.forEach(function(item){console.log(item);})
  //  add
  //  delete
  //  clear
  //  has
}

//WeakSet()只能是对象，且它是弱引用，不会检测对象是否已经存在，所以不和垃圾机制挂钩
//不支持遍历
{
  let weakList=new WeakSet();
  let arg={};
  weakList.add(arg);
  // weakList.add(2); //报错
  console.log('weakList',weakList); //{Object {}}
}

//不带参数的定义
{
  let map = new Map();
  let arr=['123'];
  //Set用add添加元素，Map用set添加元素
  map.set(arr,456);
  console.log('map',map,map.get(arr)); //{["123"] => 456} 456
}
//带参数的定义
{
  let map = new Map([['a',123],['b',456]]);
  console.log('map args',map); //{"a" => 123, "b" => 456}
  console.log('size',map.size); //2
  console.log('delete',map.delete('a'),map); //true {"b" => 456}
  console.log('clear',map.clear(),map); //{}
}

//特性和WeakSet()差不多
{
  let weakmap=new WeakMap();
  let o={};
  weakmap.set(o,123);
  console.log(weakmap,weakmap.get(o)); //{{} => 123} 123
}
