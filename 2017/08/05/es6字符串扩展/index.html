发表于 2017-08-05 | 分类于 es6 
let s='𠮷';
console.log('length',s.length); //2
// 一个字母是两个字节，一个汉子大于两个字节被处理为4个字节，每两个字节长度为1，所以结果是2

{
  //es5
  let s='𠮷';
  console.log('length',s.length);  //length 2
  console.log('0',s.charAt(0));    // 0 �
  console.log('1',s.charAt(1));    // 1 �
  console.log('at0',s.charCodeAt(0)); // at0 55362  charCodeAt取的是两个字节的码值
  console.log('at1',s.charCodeAt(1)); // at1 57271
  //es6  
  let s1='𠮷a';
  console.log('length',s1.length);       // length 3
  console.log('code0',s1.codePointAt(0)); // code0 134071  codePointAt取的是四个字节的码值
  console.log('code0',s1.codePointAt(0).toString(16)); // code0 20bb7 toString(16)转成16进制
  console.log('code1',s1.codePointAt(1));  //code1 57271  取的是前四个的后两个字节
  console.log('code2',s1.codePointAt(2));  //code2 97   继续向后取两个
}

{
  //es5
  console.log(String.fromCharCode("0x20bb7"));   // ஷ 乱码
  //es6
  console.log(String.fromCodePoint("0x20bb7"));  // 𠮷
}
