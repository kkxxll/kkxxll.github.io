es6正则扩展

发表于 2017-08-04 | 分类于 es6 
{
  //es5
  let regex1 = new RegExp('abc', 'i'); //第一个参数是字符串，第二个是修饰符
  let regex2 = new RegExp(/abc/i);  //第一个参数是正则表达式，不接受第二个参数，否则会报错
  console.log(regex1.test('abc123'),regex2.test('abc123'));     //true true

  //es6
  let regex3 = new RegExp(/abc/ig, 'i');//原有正则对象的修饰符是ig，它会被第二个参数i覆盖

  console.log(regex3.test('abc123'));                           //true
  console.log(regex3.flags)                                     //i
}
